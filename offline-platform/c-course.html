Learn C - A powerful and efficient programming language for system programming

What is C?

C is a general-purpose, procedural programming language created in the 1970s. It's one of the most influential programming languages and forms the foundation for many modern languages.

Why Learn C?

  • Foundation for understanding computer science concepts
  • Used in operating systems, embedded systems, and databases
  • Teaches important programming fundamentals
  • Efficient and close to hardware operations
  • Still widely used in industry

C Program Structure

Every C program follows a basic structure:

#include <stdio.h> int main() { printf("Hello, World!\n"); return 0; }

Key Components:

  • #include <stdio.h> - Include input/output library
  • int main() - Entry point of program
  • printf() - Print function (output)
  • return 0; - Return success status

Common Header Files

#include <stdio.h> // Input/Output #include <stdlib.h> // Standard library #include <string.h> // String functions #include <math.h> // Math functions #include <ctype.h> // Character functions #include <time.h> // Time functions

Variables and Data Types

C requires explicit type declaration for all variables:

#include <stdio.h> int main() { // Integer types int age = 25; // Standard integer short x = 10; // Small integer long population = 7800000000; // Floating point float price = 19.99; double pi = 3.14159; // Character char initial = 'A'; // String (array of chars) char name[50] = "John"; // Print values printf("Age: %d\n", age); printf("Price: %.2f\n", price); printf("Name: %s\n", name); return 0; }

Data Type Sizes

#include <stdio.h> int main() { printf("Size of int: %lu bytes\n", sizeof(int)); printf("Size of float: %lu bytes\n", sizeof(float)); printf("Size of double: %lu bytes\n", sizeof(double)); printf("Size of char: %lu bytes\n", sizeof(char)); return 0; }

Input and Output (I/O)

Essential for communicating with users:

printf() - Output

#include <stdio.h> int main() { int age = 25; float height = 5.9; printf("Simple string\n"); printf("Age: %d\n", age); printf("Height: %.1f feet\n", height); printf("Age: %d, Height: %.1f\n", age, height); return 0; } /* Format specifiers: %d - Integer %f - Float (default 6 decimals) %.2f - Float with 2 decimals %s - String %c - Character %x - Hexadecimal %o - Octal */

scanf() - Input

#include <stdio.h> int main() { int age; float height; char name[50]; printf("Enter your name: "); scanf("%s", name); printf("Enter your age: "); scanf("%d", &age); printf("Enter your height: "); scanf("%f", &height); printf("Name: %s, Age: %d, Height: %.1f\n", name, age, height); return 0; }

Operators

C supports various operators for calculations and logic:

Arithmetic Operators

#include <stdio.h> int main() { int a = 10, b = 3; printf("Addition: %d\n", a + b); // 13 printf("Subtraction: %d\n", a - b); // 7 printf("Multiplication: %d\n", a * b);// 30 printf("Division: %d\n", a / b); // 3 printf("Modulo: %d\n", a % b); // 1 return 0; }

Comparison Operators

a == b // Equal a != b // Not equal a > b // Greater than a < b // Less than a >= b // Greater or equal a <= b // Less or equal

Logical Operators

a && b // AND a || b // OR !a // NOT

Conditional Statements

Make decisions in your code:

if...else Statement

#include <stdio.h> int main() { int age = 18; if (age >= 18) { printf("You are an adult\n"); } else { printf("You are a minor\n"); } return 0; }

if...else if...else

#include <stdio.h> int main() { int score = 85; if (score >= 90) { printf("Grade: A\n"); } else if (score >= 80) { printf("Grade: B\n"); } else if (score >= 70) { printf("Grade: C\n"); } else { printf("Grade: F\n"); } return 0; }

switch Statement

#include <stdio.h> int main() { int day = 3; switch (day) { case 1: printf("Monday\n"); break; case 2: printf("Tuesday\n"); break; case 3: printf("Wednesday\n"); break; default: printf("Unknown day\n"); } return 0; }

Loops

Repeat code blocks efficiently:

for Loop

#include <stdio.h> int main() { // Print numbers 1 to 5 for (int i = 1; i <= 5; i++) { printf("%d\n", i); } return 0; }

while Loop

#include <stdio.h> int main() { int i = 1; while (i <= 5) { printf("%d\n", i); i++; } return 0; }

do...while Loop

#include <stdio.h> int main() { int i = 1; do { printf("%d\n", i); i++; } while (i <= 5); // Executes at least once return 0; }

Loop Control

for (int i = 1; i <= 10; i++) { if (i == 3) { continue; // Skip this iteration } if (i == 8) { break; // Exit loop } printf("%d\n", i); }

Functions

Reusable code blocks with specific purposes:

Function Definition

#include <stdio.h> // Function declaration int add(int a, int b); int main() { int result = add(5, 3); printf("Sum: %d\n", result); return 0; } // Function definition int add(int a, int b) { return a + b; }

Functions with No Return Value

#include <stdio.h> void greet(char name[]) { printf("Hello, %s!\n", name); } int main() { greet("Alice"); return 0; }

Function Example - Largest Number

#include <stdio.h> int findMax(int a, int b, int c) { int max = a; if (b > max) max = b; if (c > max) max = c; return max; } int main() { int result = findMax(10, 25, 15); printf("Maximum: %d\n", result); return 0; }

Arrays

Store multiple values in a single variable:

Array Basics

#include <stdio.h> int main() { // Declare and initialize int numbers[5] = {10, 20, 30, 40, 50}; // Access elements printf("First element: %d\n", numbers[0]); printf("Third element: %d\n", numbers[2]); // Modify element numbers[1] = 25; // Loop through array for (int i = 0; i < 5; i++) { printf("%d\n", numbers[i]); } return 0; }

String as Character Array

#include <stdio.h> #include <string.h> int main() { char name[50] = "John"; // String functions printf("Length: %lu\n", strlen(name)); // String comparison if (strcmp(name, "John") == 0) { printf("Name matches\n"); } // String copy char name2[50]; strcpy(name2, name); return 0; }

Pointers

A powerful feature to work with memory addresses:

#include <stdio.h> int main() { int age = 25; int *ptr; // Declare pointer ptr = &age; // & gets address of variable printf("Value of age: %d\n", age); printf("Address of age: %p\n", &age); printf("Pointer value: %p\n", ptr); printf("Value via pointer: %d\n", *ptr); // * dereference // Modify through pointer *ptr = 30; printf("New age: %d\n", age); return 0; }

Key Concepts:

  • & (Address operator) - Gets the address of a variable
  • * (Dereference operator) - Gets the value at an address
  • Pointers store memory addresses
  • Used for dynamic memory, functions that modify variables, and arrays
